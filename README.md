```bash
src
  - components # 公用组件
  - constants # 常量
  - helpers # 当前项目内的一些帮助方法
  - hooks # 公用 hooks
  - modules # 项目模块
    - hooks # 当前模块内 hooks
    - components # 当前模板内 components
    - helpers # 当前模板内 helpers
    - reducer # 当前模板内的 reducer
  - pages # 单独的页面，如404、登录页面
  - services # 公用服务数据处理
  - utils # 一些通用方法，可跨项目使用
```
